vpop_seffect_add_poptype = {
	scope:p = {
		switch = {
			trigger = is_pop_type
			academics   = { prev = { change_variable = { name = vpop_Academics   add = $N$ } }  }
			aristocrats = { prev = { change_variable = { name = vpop_Aristocrats add = $N$ } }  }
			bureaucrats = { prev = { change_variable = { name = vpop_Bureaucrats add = $N$ } }  }
			capitalists = { prev = { change_variable = { name = vpop_Capitalists add = $N$ } }  }
			clergymen   = { prev = { change_variable = { name = vpop_Clergymen   add = $N$ } }  }
			clerks      = { prev = { change_variable = { name = vpop_Clerks      add = $N$ } }  }
			engineers   = { prev = { change_variable = { name = vpop_Engineers   add = $N$ } }  }
			farmers     = { prev = { change_variable = { name = vpop_Farmers     add = $N$ } }  }
			laborers    = { prev = { change_variable = { name = vpop_Laborers    add = $N$ } }  }
			machinists  = { prev = { change_variable = { name = vpop_Machinists  add = $N$ } }  }
			officers    = { prev = { change_variable = { name = vpop_Officers    add = $N$ } }  }
			peasants    = { prev = { change_variable = { name = vpop_Peasants    add = $N$ } }  }
			shopkeepers = { prev = { change_variable = { name = vpop_Shopkeepers add = $N$ } }  }
			slaves      = { prev = { change_variable = { name = vpop_Slaves      add = $N$ } }  }
			soldiers    = { prev = { change_variable = { name = vpop_Servicemen  add = $N$ } }  }
		}
	}
}
