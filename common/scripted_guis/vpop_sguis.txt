vpop_sgui_test_set2 = {
	scope = country

	effect = {
		set_global_variable = {
			name = vpop_Recalc
			days = 1
		}
		set_global_variable = {
			name = vpop_BuildingID
			value = scope:id
		}
		vpop_effect_clear_values = yes
	}
}

vpop_sgui_add_poptype_interest_group = {
	scope = interest_group

	effect = {
		if = {
			limit = {
				scope:p = {
					or = {
						is_pop_type = slaves
						pop_is_discriminated = yes
					}
				}
			}
			owner = {
				vpop_seffect_add_poptype = yes
			}
		}
		else = {
			vpop_seffect_add_poptype = yes
		}
	}
}

vpop_sgui_add_poptype_country = {
	scope = country

	effect = {
		vpop_seffect_add_poptype = yes

		scope:p.culture = {
			vpop_seffect_add_poptype = yes
		}
	}
}
