vpop_sgui_test_set2 = {
	scope = country

	effect = {
		set_global_variable = {
			name = vpop_Recalc
			days = 1
		}
		set_global_variable = {
			name = vpop_BuildingID
			value = scope:id
		}
		vpop_seffect_clear_poptypes = yes
		every_interest_group = {
			vpop_seffect_clear_poptypes = yes
		}
		every_scope_culture = {
			vpop_seffect_clear_poptypes = yes
		}

	}
}

vpop_sgui_add_poptype_interest_group = {
	scope = interest_group

	effect = {
		if = {
			limit = {
				scope:p = {
					or = {
						is_pop_type = slaves
						pop_is_discriminated = yes
					}
				}
			}
			owner = {
				vpop_seffect_add_poptype = {
					N = scope:n
				}
			}
		}

		else = {
			vpop_seffect_add_poptype = {
				N = scope:n
			}
		}
	}
}

vpop_sgui_add_poptype_country = {
	scope = country

	effect = {
		vpop_seffect_add_poptype = {
			N = scope:u
		}

		scope:p.culture = {
			vpop_seffect_add_poptype = {
				N = scope:n
			}
		}
	}
}
